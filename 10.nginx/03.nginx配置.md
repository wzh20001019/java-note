#### 1. events 网络连接相关配置

```conf
events {
    # 默认最大并发数
    worker_connections  1024;
}
```

#### 2. http 代理,缓存,日志,虚拟机...

```conf
http {
    include       mime.types;
    default_type  application/octet-stream;

    # log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
    #                   '$status $body_bytes_sent "$http_referer" '
    #                   '"$http_user_agent" "$http_x_forwarded_for"';

    # access_log  logs/access.log  main;

    sendfile        on;
    # tcp_nopush     on;

    # keepalive_timeout  0;
    keepalive_timeout  65;

    # gzip  on;
    server {
        listen       9000; # 监听端口
        server_name  127.0.0.1; # 服务地址

        # charset koi8-r;

        # access_log  logs/host.access.log  main;

        # upstream配置一组服务器(负载均衡)  weight(分发权重)  ...
        upstream targetServer {
            server 192.168.100.105:8000 weight=10
            server 127.0.0.1:3000 weight=5
            server 103.29.34.79:8080 weight=1
        }

        # 匹配客服端 url 请求 (配置静态资源)
        location /cpadmin {
            # 连接服务器超时时间设置单位S
            proxy_connect_timeout 10;
            proxy_read_timeout 10;
            proxy_send_timeout 10;

            # 指定静态资源目录
            root   E:/Front;
            try_files $uri $uri/ /cpadmin/index.html;

            # 指定默认首页
            index  index.html index.htm;
        }

        location /statics {
            root E:/Front;
            if (-f $request_filename) {
                expires max;
                break;
            }
        }

        # 配置反向代理
        location /bocca {
            # 连接服务器超时时间设置单位S
            proxy_connect_timeout 10;
            proxy_read_timeout 10;

            # 代理地址 (加/bocca后缀)
            # proxy_pass http://boccaccio.3322.org:22202;

            # 代理地址 (不加/bocca后缀)
            proxy_pass http://boccaccio.3322.org:22202/;
        }

        location /new {
            # 连接服务器超时时间设置单位S
            proxy_connect_timeout 10;
            proxy_read_timeout 10;

            # 负载均衡
            proxy_pass http://targetServer;
        }


        # error_page  404              /404.html;

        # redirect server error pages to the static page /50x.html
        #
        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
    }
}
```
