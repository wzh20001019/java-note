#### 1. 部署 mysql

```txt
// 1. 下载
docker pull mysql:版本

// 2. home下 创建mysql文件夹
mkdir mysql
cd mysql

// 3. 配置    $PWD(当前文件夹)
docker run -id \
-p 3306:3306 \
--name=my_mysql \
-v /home/mysql/conf:/etc/mysql/conf.d \
-v /home/mysql/logs:/logs \
-v /home/mysql/data:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=123456 \
mysql:5.7

// 4. 进入容器
docker exec -it my_mysql /bin/bash
```

#### 2. 部署 tomcat

```txt
// 1. 下载
docker pull tomcat

// 2. home下 创建tomcat文件夹
mkdir tomcat
cd tomcat

// 3. 配置
docker run -id \
--name=my_tomcat \
-p 8080:8080 \
-v /home/tomcat:/user/local/tomcat/webapps \
tomcat

// 4. 进入容器
docker exec -it my_tomcat /bin/bash
```

#### 3. 部署 nginx

```txt
// 1. 下载
docker pull nginx

// 2. home下 创建nginx文件夹
mkdir nginx
cd nginx

mkdir conf
cd conf
touch nginx.conf

// 配置 nginx.conf
vim nginx.conf

// 3. 配置
docker run -id \
--name=my_nginx \
-p 80:80 \
-v /home/nginx/conf/nginx.conf:/etc/nginx/nginx.conf \
-v /home/nginx/logs:/var/log/nginx \
-v /home/nginx/html:/user/share/nginx/html \
nginx

// 4. 进入容器
docker exec -it my_nginx /bin/bash
```
