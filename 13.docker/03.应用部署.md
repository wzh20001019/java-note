#### 1. 部署 mysql

```txt
// 1. 下载
docker pull mysql:版本

// 2. home下 创建mysql文件夹
mkdir mysql
cd mysql

// 3. 配置    $PWD(当前文件夹)
docker run -id \
--restart=always \
-p 3306:3306 \
--name=my_mysql \
-v /home/mysql/conf:/etc/mysql/conf.d \
-v /home/mysql/logs:/logs \
-v /home/mysql/data:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=123456 \
mysql:5.7

// 4. 进入容器
docker exec -it my_mysql /bin/bash
```

#### 2. 部署 tomcat

```txt
// 1. 下载
docker pull tomcat

// 2. home下 创建tomcat文件夹
mkdir tomcat
cd tomcat

// 3. 配置
docker run -id \
--restart=always \
--name=my_tomcat \
-p 8080:8080 \
-v /home/tomcat:/user/local/tomcat/webapps \
tomcat

// 4. 进入容器
docker exec -it my_tomcat /bin/bash
```

#### 3. 部署 nginx

```txt
// 1. 下载
docker pull nginx

// 2. 宿主机创建好要挂载的目录
mkdir -p /home/nginx
mkdir -p /home/nginx/www
mkdir -p /home/nginx/conf
mkdir -p /home/nginx/logs

// 3. 启动一个不挂载的容器
docker run -id --name=my_nginx  -p 80:80 nginx

// 4. 从容器中把配置文件复制出来
docker cp my_nginx:/etc/nginx/nginx.conf /home/nginx/
docker cp my_nginx:/etc/nginx/conf.d /home/nginx/conf/
docker cp my_nginx:/usr/share/nginx/html/ /home/nginx/www/
docker cp my_nginx:/var/log/nginx/ /home/nginx/logs/

// 5. 停止并移除容器
docker stop my_nginx            停止容器
docker rm -f my_nginx           移除容器

// 6. 应用部署
docker run -id \
--restart=always \
--name=my_nginx \
-p 80:80 \
-v /home/nginx/nginx.conf:/etc/nginx/nginx.conf \
-v /home/nginx/conf/conf.d/:/etc/nginx/conf.d \
-v /home/nginx/www/html:/usr/share/nginx/html/ \
-v /home/nginx/logs/:/var/log/nginx/ \
--privileged=true \
nginx

// 7. 进入容器
docker exec -it my_nginx /bin/bash

// 8. 项目复制到容器中
docker cp 本地文件路径 容器ID/容器NAME:容器内路径
```
