#### 1. 部署 mysql

```txt
// 1. 下载
docker pull mysql:版本

// 2. home下 创建mysql文件夹
mkdir mysql
cd mysql

// 3. 配置    $PWD(当前文件夹)
docker run -id \
--restart=always \
-p 3306:3306 \
--name=my_mysql \
-v /home/mysql/conf:/etc/mysql/conf.d \
-v /home/mysql/logs:/logs \
-v /home/mysql/data:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=123456 \
mysql:5.7

// 4. 进入容器
docker exec -it my_mysql /bin/bash
```

#### 2. 部署 tomcat

```txt
// 1. 下载
docker pull tomcat

// 2. home下 创建tomcat文件夹
mkdir tomcat
cd tomcat

// 3. 配置
docker run -id \
--restart=always \
--name=my_tomcat \
-p 8080:8080 \
-v /home/tomcat:/user/local/tomcat/webapps \
tomcat

// 4. 进入容器
docker exec -it my_tomcat /bin/bash
```

#### 3. 部署 nginx

```txt
// 1. 下载
docker pull nginx

// 2. 宿主机创建好要挂载的目录
mkdir -p /home/nginx/
mkdir -p /home/nginx/logs
mkdir -p /home/nginx/html

// 3. 启动一个不挂载的容器
docker run -d --name my_nginx  -p 3000:3000 nginx

// 4. 从容器中把配置文件复制出来
docker cp nzc-nginx:/etc/nginx/nginx.conf /home/nginx/nginx.conf
docker cp nzc-nginx:/etc/nginx/conf.d /home/nginx/
docker cp nzc-nginx:/usr/share/nginx/html /home/nginx/ #此处就是网站站点目录

// 3. 配置
docker run -id \
--restart=always \
--name=my_nginx \
-p 3000:3000 \
-v /home/nginx/conf/nginx.conf:/etc/nginx/nginx.conf \
-v /home/nginx/logs:/var/log/nginx \
-v /home/nginx/html:/user/share/nginx/html \
nginx

// 4. 进入容器
docker exec -it my_nginx /bin/bash
```
